# Import package
from urllib.request import urlretrieve

# Import pandas
import pandas as pd

# Assign url of file: url

# Save file locally
urlretrieve(url,"winequality-red.csv")

# Read file into a DataFrame and print its head
df = pd.read_csv('winequality-red.csv', sep=';')
print(df.head())
