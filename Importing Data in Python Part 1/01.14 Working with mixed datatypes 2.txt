# Assign the filename: file
file = 'titanic.csv'

# Import file using np.recfromcsv: d
d = np.recfromcsv(file, delimiter=",",names=True,dtype=None)

# Print out first three entries of d
print(d[:3])
print(d.dtype.names)
print(d.shape)
